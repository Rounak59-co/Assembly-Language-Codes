LDI R25, 20 ; R25 WILL HOLD THE VALUE OF REQUIRED NUMBER

LDI R26,0X00
LDI R27,0X01   ; THE STARTING LOCATION OF ARRAY IS STORED IN X

LDI R28,0X07
LDI R29,0X01   ; THE END LOCATION OF ARRAY IS STORED IN Y

MOV R21,R26   ; USED LATER TO COMPARE R26 WITH IT'S INITIAL VALUE

BACK:
	MOV R19,R28   ; THS PART IS FOR EDGE CASES
	DEC R19
	CP R19,R26
	BREQ EQUAL
	MOV R16,R26   ; HERE WE WILL START COMPARING. VALUE OF R26 WILL BE STORED IN R16 TO USE R26 AS LOWER LIMIT
	ADD R26,R28   ; LOWER LIMIT + UPPER LIMIT
	LSR R26       ; (L+H)/2
	LD R17,X      ; THE MID VALUE IS STORED IN R17
	CP R17,R25    ; IT IS COMPARED WITH REQUIRED NUMBER
	BRLO BACK     ; IF REQUIRED NUMBER > MID VALUE , THEN MID POSITION WILL BECOME LOWER LIMIT 
	CPSE R25,R17  ; HERE, WE COMPARE IF MID VALUE >= REQUIRED NUMBER
	RJMP SMALL    ; IF >, THEN JUMP TO SMALL
	RJMP END      ; IF =, THE INDEX WILL STAY STORED IN R26

SMALL:
	MOV R28,R26    ; WE SET MID POSITION AS UPPER LIMIT
	MOV R26,R16    ; RESTORE THE R26 TO INITIAL POSITION
 	RJMP BACK       
EQUAL:                      ; THIS PART IS FOR EDGE VALUES
CP R26,R21                  ; THIS ONE IS FOR LEFT EDGE VALUE
BREQ END
LD R20,Y                    ; THIS ONE IS FOR RIGHT EDGE VALUE 
CP R20,R25
BRNE ERROR
INC R26
RJMP END
ERROR:
LDI R26,0XEE  ;ERR IS FOR INVALID VALUES
END:RJMP END  ; THE INDEX OF THE REQUIRED NUMBER WILL GET STORED IN R26
